## 项目结构
```
MaaYYs/
├───assets  # 资源文件夹
│   ├───config  # 配置文件夹
│   │   ├── app_config.json  # app客户端配置
│   │   ├── maa_option.json  # maa客户端配置
│   │   ├── programs.json    #app端 各项程序(任务)配置
│   │   └── projects.json    #app端 各项项目配置
│   ├───icons
│   │   └───svg_icons       #图标文件夹
│   ├───MaaCommonAssets     #MAAFW通用资源文件夹(暂时还没啥用)
│   │   └───OCR
│   │       ├───ppocr_v3
│   │       │   ├───en_us
│   │       │   ├───ja_jp
│   │       │   ├───ko_kr
│   │       │   ├───zh_cn
│   │       │   └───zh_tw
│   │       └───ppocr_v4
│   │           ├───en_us
│   │           └───zh_cn
│   ├───myrce               #myrce资源文件夹
│   └───resource            #主要任务资源文件夹(详细请看MAAFW)
│       └───base            #基础资源文件夹
│           ├───image       #MAAFW匹配图片文件夹   
│           ├───model       #MAAFW模型文件夹
│           │   └───ocr
│           └───pipeline    #MAAFW主要任务文件夹
├───DLL                     #DLL文件夹(暂时主要用于修复python11环境下的问题)
├───doc                     #文档文件夹
│   └───doc_img
└───src                     #源码文件夹
    ├───service             #服务文件夹
    ├───ui                  #程序主要UI
    └───utils               #工具文件夹

```
